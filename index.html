import os import sqlite3 from flask import Flask, render_template_string, request, redirect, url_for, session, flash from werkzeug.security import generate_password_hash, check_password_hash from datetime import datetime app = Flask(__name__) app.secret_key = os.urandom(24) DB_PATH = 'reports.db' ADMIN_USERNAME = 'admin' ADMIN_PASSWORD_HASH = generate_password_hash('admin') # --- Инициализация базы данных --- def init_db(): conn = sqlite3.connect(DB_PATH) c = conn.cursor() c.execute(''' CREATE TABLE IF NOT EXISTS reports ( id INTEGER PRIMARY KEY AUTOINCREMENT, title TEXT NOT NULL, description TEXT, photo_path TEXT, status TEXT DEFAULT 'new', created_at TEXT NOT NULL, updated_at TEXT NOT NULL ) ''') c.execute(''' CREATE TABLE IF NOT EXISTS users ( username TEXT PRIMARY KEY, password_hash TEXT NOT NULL ) ''') c.execute('INSERT OR IGNORE INTO users (username, password_hash) VALUES (?, ?)', (ADMIN_USERNAME, ADMIN_PASSWORD_HASH)) conn.commit() conn.close() init_db() # --- Вспомогательные функции --- def get_reports(status=None): conn = sqlite3.connect(DB_PATH) conn.row_factory = sqlite3.Row c = conn.cursor() if status: c.execute('SELECT * FROM reports WHERE status=? ORDER BY created_at DESC', (status,)) else: c.execute('SELECT * FROM reports ORDER BY created_at DESC') rows = c.fetchall() conn.close() return [dict(row) for row in rows] def get_report(report_id): conn = sqlite3.connect(DB_PATH) conn.row_factory = sqlite3.Row c = conn.cursor() c.execute('SELECT * FROM reports WHERE id=?', (report_id,)) row = c.fetchone() conn.close() return dict(row) if row else None # --- Шаблоны (встроены в код) --- BASE_TEMPLATE = ''' Портал Наш город

''', (title, description, photo_path, now, now)) conn.commit() conn.close() flash('Отчёт успешно добавлен! Спасибо.', 'success') return redirect(url_for('index')) return render_template_string(ADD_TEMPLATE) @app.route('/login', methods=['GET', 'POST']) def login(): username = None password = None if request.method == 'POST': username = request.form.get('username') password = request.form.get('password') conn = sqlite3.connect(DB_PATH) c = conn.cursor() c.execute('SELECT password_hash FROM users WHERE username=?', (username,)) row = c.fetchone() conn.close() if row and check_password_hash(row[0], password): session['logged_in'] = True return redirect(url_for('admin')) flash('Неверный логин или пароль.', 'danger') return render_template_string(LOGIN_TEMPLATE, username=username, password=password) @app.route('/logout') def logout(): session.pop('logged_in', None) return redirect(url_for('index')) def admin_required(f): def wrap(*args, **kwargs): if not session.get('logged_in'): return redirect(url_for('login')) return f(*args, **kwargs) wrap.__name__ = f.__name__ return wrap @app.route('/admin') @admin_required def admin(): new_reports = get_reports(status='new') in_progress_reports = get_reports(status='in_progress') done_reports = get_reports(status='done') return render_template_string( ADMIN_TEMPLATE, new_reports=new_reports, in_progress_reports=in_progress_reports, done_reports=done_reports, ) @app.route('/admin/report//update_status', methods=['POST']) @admin_required def update_status(report_id): status = request.form.get('status') now = datetime.now().strftime('%Y-%m-%d %H:%M:%S') conn = sqlite3.connect(DB_PATH) c = conn.cursor() c.execute('UPDATE reports SET status=?, updated_at=? WHERE id=?', (status, now, report_id)) conn.commit() conn.close() flash('Статус обновлён.', 'success') return redirect(url_for('admin')) if name == '__main__': app.run(debug=True)
